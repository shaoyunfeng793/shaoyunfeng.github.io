Webpack之抽离压缩css

2022/9/30 Webpack

Webpack之抽离压缩css

使用 mini-css-extract-plugin 插件的内置loader 替换 style-loader

使用 terser-webpack-plugin 插件压缩js代码

使用 optimize-css-assets-webpack-plugin 插件压缩css代码

Webpack 配置

    const MiniCSSExtractPlugin = require('mini-css-extract-plugin');
 
   
    const TerserPlugin = require('terser-webpack-plugin');  // 压缩js插件
    const OptimizeCssAssetsWebpackPlugin = require('optimize-css-assets-webpack-plugin');  // 压缩css插件
    
    module.exports = {
        // ...
        module: {
            rules: [
                {
                    test: /\.css/,
                    loader: [
                        MiniCSSExtractPlugin.loader, // 替换 style-loader
                        'css-loader',
                        'postcss-loader',  // autoprefixer
                    ],
                }
            ]
        },
        plugins: [
            new MiniCSSExtractPlugin({
                filename: 'css/main.[contentHash:8].css'
            })
        ],
        // 优化配置
        optimization: {
            minimizer: {
                // 压缩js
                new TerserPlugin(),
                // 压缩css
                new OptimizeCssAssetsWebpackPlugin()
            }
        }  
    }
