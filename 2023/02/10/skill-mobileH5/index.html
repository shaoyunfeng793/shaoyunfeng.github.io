vueWEB项目打包成app

使用Hbuildx工具打包成5+app即可

1.在hbuildx里先新建一个项目（5+app）
2.在vue项目中先build一下，生成到dist目录
3.将dist目录下的代码全都拷贝到新建的5+app目录下，直接覆盖即可
4.可以点击运行到浏览器进行预览
5.云端打包成app需要的安卓证书、ios证书获取方式见文档

安卓端侧边滑动返回监听

把下面代码直接放在public下面的index.html的body下面即可

    <script>
        const webview = plus.webview.currentWebview();
        document.addEventListener('plusready', () => {
            //等待plus ready后再调用5+ API：
            let first = null;
            plus.key.addEventListener('backbutton', () => {

                webview.canBack((e) => {
                    if (e.canBack) {
                        webview.back();
                    } else {
                        //首次按键，提示‘再次操作退出应用’
                        if (!first) {
                            //获取第一次点击的时间戳
                            first = new Date().getTime();
                            //通过H5+ API 调用Android 上的toast 提示框
                            plus.nativeUI.toast("再次操作退出应用", {
                                duration: 'short'
                            });
                            setTimeout(() => {
                                first = null;
                            }, 1000);
                        } else {
                            if (new Date().getTime() - first < 1000) {
                                //获取第二次点击的时间戳, 两次之差 小于 1000ms 说明1s点击了两次,
                                plus.runtime.quit(); //退出应用
                            }
                        }
                    }
                })
            });
        });
    </script>

由于vue项目打包成app其实本身就是套在webview里面运行的，所以一些跳转页面除了$router.push还有就是location.href，如果不兼容一下APP里的会导致页面返回不了，举个例子：app里有个广告banner,点击跳转http链接，结果用户侧滑会直接导致退出应用，即关闭掉了webview，因为app页面承载本身就是一个webviewObject,跳转后的http链接是直接在这个webviewObject上显示的，通过下面代码帮我们兼容跳转http(可以控制app内部打开/外部浏览器打开)

function toWebLink(url, openType = 'outside') {
  try {
    if (openType === 'inner') {
      let wv = plus.webview.getWebviewById('webview');
      // plus api 创建并打开新的webview 详见：https://www.html5plus.org/doc/zh_cn/webview.html#plus.webview.open
      plus.webview.open(
        url,
        'webview',
        {
          titleNView: {
            titleSize: '14px',
            buttons: [
              {
                type: 'back',
                float: 'left',
                onclick: function () {
                  wv.canBack((e) => {
                    if (e.canBack) {
                      wv.back();
                    } else {
                      wv.close();
                    }
                  });
                },
              },
            ],
          },
        },
        'slide-in-right',
        300,
        function () {
          // console.log('打开了', JSON.stringify(plus.webview.all()));
        },
      );
    } else {
      plus.runtime.openURL(url);
    }

  } catch (e) {
    console.log(e);
    window.location.href = url;
  }
}
