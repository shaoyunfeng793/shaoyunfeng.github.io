BOM：history的那些事

前言

现在前端框架都有路由组件，路由包括2种模式：hash、history

而前端路由实现的原理好像和Bom对象里的history有点联系…

Emm…瞎搞一下…

一、Hash模式

hash模式下，通过location来跳转，才会触发hashchange监听事件， 再通过lcoation.hash不同的hash值来决定渲染的组件, 路由的hash模式其工作方式类似

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>hash模式</title>
</head>

<body>
    <button onclick="turn('/#/abc')">跳转到：/#/abc</button>
    <div id="view"></div>
</body>
<script>
    function turn(url) {
        
        let state = {
            a: 1,
            b: 2
        }
        // hash模式下，通过locaiton来跳转
        location.href = url;
        // history.pushState(state, '/abc', url) // 该方式不会触发hashchange
    }
    // 监听url的hash变化
    window.addEventListener('hashchange', function (e) {
        console.log('旧链接：' + e.oldURL)
        console.log('新链接：' + e.newURL)
        console.log(location.hash); // => #/abc
        document.getElementById('view').innerHtml = `<div>${location.hash}组件</div>`
    })
    // pushState replaceState都不会触发,用户前进后退或者location方式才触发
    window.addEventListener('popstate', function (e){
        console.log(e)
    })
</script>

</html>

二、History模式

pushState只会改变url,不是真正的去跳转, 以下代码中跳转到 /abc ,如果使用location会404,因为没有对应的资源

但是pushState没有相应的onhistorychange事件可以监听,那就只能自己写一个push,还有historychange事件

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>history模式</title>
</head>

<body>
    <button onclick="turn('/abc')">跳转到：/abc</button>
    <div id="view"></div>
</body>
<script>
    // 当然路由也不是这么样子的, 因为懒
    var router = new Router([
        {
            name: 'abc',
            path: '/abc',
            component: 'abc组件(假装是个组件)',
        }
    ]);
    function turn(url) {

        router.push(url)
    }
    function Router(routes) {
        this.routes = routes || [];

        window.addEventListener('historychange', function (e) {
            let route = e.detail;
            console.log('改变了, 拿到的路由对象是:', route)
            // 渲染路由组件
            document.getElementById('view').innerHTML = route.component;
        })
    }
    Router.prototype.push = function (url) {
        // 这么不严谨的代码,只是因为太懒
        const findRoute = this.routes.find(d => d.path === url);

        if (findRoute) {
            const ev = new CustomEvent('historychange', { detail: findRoute });

            // history模式下，通过pushState来跳转
            history.pushState({}, null, url);
            // location.href = url; // 该方式不行，会404
            window.dispatchEvent(ev);
        } else {
            throw('该路由不存在')
        }
    }
</script>

</html>
